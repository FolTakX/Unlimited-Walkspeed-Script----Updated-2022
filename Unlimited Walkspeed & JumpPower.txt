local Walkspeed = 1000 -- Set this value to whatever you like.
local JumpPower = 320 -- Same goes with the JumpPower.


while wait(1) do --Loop the code every second
    if game.Players.LocalPlayer.Character.Humanoid.WalkSpeed ~= Walkspeed then
        game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = Walkspeed
    end
    if game.Players.LocalPlayer.Character.Humanoid.JumpPower ~= JumpPower then 
        game.Players.LocalPlayer.Character.Humanoid.JumpPower = JumpPower
    end
end
if game.Players.LocalPlayer.Character.Humanoid.WalkSpeed == Walkspeed and game.Players.LocalPlayer.Character.Humanoid.JumpPower == JumpPower then
    print("Script successfully loaded.")
else
    print("Failed to load the script, please contact the owner! Error Code: 1")
end
while wait (1) do if game.Players.LocalPlayer.Character.Humanoid.WalkSpeed ~= Walkspeed and game.Players.LocalPlayer.Character.Humanoid.JumpPower ~= JumpPower then
        while wait(1) do if game.Players.LocalPlayer.Character.Humanoid.WalkSpeed ~= Walkspeed then game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = Walkspeed end if game.Players.LocalPlayer.Character.Humanoid.JumpPower ~= JumpPower then game.Players.LocalPlayer.Character.Humanoid.JumpPower = JumpPower end end
        print("You have died, but the script reloaded.")
   end
end
